v-value-card.js 使用方式

Source Code

                
                <v-value-card 
                    id="sp1"
                    txt='銷售額'
                    charttype="line"
                    val="$simple string  or  method"
                    icon="attach_money"
                    :chartdata="chartdata"
                    chartcolor="rgba(15,219,183,0.4)"
                    linecolor="rgba(15,219,183,1)"
                    emptystyle="height: 171px"
                    :meskey="['時段', '銷售額']"
                    :y_label_format="(val, key) => {
                        if (['銷售額'].includes(key)) return '$' + val.toLocaleString()
                        else if (['時段'].includes(key)) return `${val}:00 ~ ${val}:59` 
                        else return val.toLocaleString()
                    }">
                </v-value-card>
